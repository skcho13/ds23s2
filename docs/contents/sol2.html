# 2. Is there a relationship between the age of a plane and its delays?

plane_age = (
    planes[["tailnum", "year"]]
    .merge(flights, on="tailnum", how="right", suffixes=("_plane", ""))
    .drop("tailnum", axis=1)
    .assign(age=lambda x: x.year - x.year_plane)
